# get data generators from folders
def get_generator(path = path, rotation = 45, shear_range = 0.2, width = 0.2, height = 0.2, zoom = 0.2, flip = True, fill = 'nearest', image_width = 256, image_height = 256):
    my_generator = ImageDataGenerator(rotation_range=rotation,width_shift_range=width,
        height_shift_range=height,shear_range=shear_range,
        zoom_range=zoom,horizontal_flip=flip,
        rescale=1./255,fill_mode=fill)
    train_generator = my_generator.flow_from_directory( path+'/train/', target_size=(image_width, image_height), batch_size=4, class_mode='categorical')
    # Generator for validation dataset
    valid_generator = my_generator.flow_from_directory( path+'/valid/', target_size=(image_width, image_height), batch_size=4, class_mode='categorical')
    # Generator for testing dataset
    test_generator = my_generator.flow_from_directory( path+'/test/', target_size=(image_width, image_height), batch_size=4, class_mode='categorical')
    return train_generator, valid_generator, test_generator

# to plot learning curve with y-axis of accuracy and x-axis of epoch
def accuracy_learning_curve(history):
    plt.plot(history.history['accuracy'])
    plt.plot(history.history['val_accuracy'])
    plt.title('Model Accuracy Learning Curve')
    plt.ylabel('Accuracy')
    plt.xlabel('Epoch')
    plt.legend(['training data', 'validation data'], loc='lower left')
    plt.show()


# to plot learning curve with y-axis of loss and x-axis of epoch
def loss_learning_curve(history):
    plt.plot(history.history['loss'])
    plt.plot(history.history['val_loss'])
    plt.title('Model Loss Learning Curve')
    plt.ylabel('Loss')
    plt.xlabel('Epoch')
    plt.legend(['training data', 'validation data'], loc='lower left')
    plt.show()

# use vgg16 model to train my dataset
def use_vgg16(train_generator,valid_generator, test_generator,epoch = 30, batch=64, dropout = 0.25,image_width = 256, image_height = 256):
    vgg16_model = VGG16(weights='imagenet', include_top=False, input_shape=(image_width, image_height , 3))
    for layer in vgg16_model.layers:
        layer.trainable = False 

    top_model = Sequential()
    top_model.add(Flatten(input_shape=vgg16_model.output_shape[1:]))
    # top_model.add(Dense(32, activation='relu'))
    top_model.add(Dropout(dropout))
    top_model.add(Dense(5, activation='softmax'))

    model = Sequential()
    model.add(vgg16_model)
    model.add(top_model)
    print(model.summary())
    early_stopping = EarlyStopping(monitor='val_accuracy', patience=8, min_delta=0.001, mode='max')
    model.compile(optimizer='adam', loss='categorical_crossentropy', metrics=['accuracy'])
    start_time = time.time()

    history = model.fit(train_generator, validation_data = valid_generator, epochs = epoch, batch_size = batch,callbacks=[early_stopping], verbose=1)
    score = model.evaluate_generator( test_generator, verbose=0)
    print(f'Test Generator loss: {score[0]} / Test Generator accuracy: {score[1]}')

    print("--- Totally needs %s seconds ---" % (time.time() - start_time))
    return history

def test_augmentation_vgg16(rotation = 90, shear_range = 0.6, width = 0.2, height = 0.2, zoom = 0.4, flip = True, fill = 'nearest', dropout = 0.5):
    train_generator, valid_generator, test_generator = get_generator(path = path, rotation = rotation, shear_range = shear_range, width = width, height = height, zoom = zoom, flip = flip, fill = fill) 
    
    history = use_vgg16( train_generator = train_generator,valid_generator = valid_generator, test_generator = test_generator, dropout = dropout)
    accuracy_learning_curve(history)
    loss_learning_curve(history)
